
 arguments for MangaScrapp :

 -a --add 'manga name' [manga site (default is manga fox)]
    add manga to database

 -u --update [manga name]
    update database / manga

 -dl --download 'manga name' [manga site (default is manga fox)]
    add manga and download

 -l --list [manga name / manga site]
    shows content of database

 -d --delete [manga name]
    deletes manga from database ( not the files )

 -h --help
    displays help

 Warning !
 The [manga name] is found in the url
 ex : "black clover" => http://mangafox.me/manga/black_clover/
 [manga name] = "black_clover"
